115  static uint16_t toFloat16(float value)
116  {
117  Bits v, s;
118  v.f = value;
119  uint32_t sign = (uint32_t) (v.si & signN);
120  v.si ^= sign;
121  sign >>= shiftSign; // logical shift
122  s.si = mulN;
123  s.si = (int32_t) (s.f * v.f); // correct subnormals
124  v.si ^= (s.si ^ v.si) & -(minN > v.si);
125  v.si ^= (infN ^ v.si) & -((infN > v.si) & (v.si > maxN));
126  v.si ^= (nanN ^ v.si) & -((nanN > v.si) & (v.si > infN));
127  v.ui >>= shift; // logical shift
128  v.si ^= ((v.si - maxD) ^ v.si) & -(v.si > maxC);
129  v.si ^= ((v.si - minD) ^ v.si) & -(v.si > subC);
130  return (uint16_t) (v.ui | sign);
131  }

133  static float toFloat32(uint16_t value)
134  {
135  Bits v;
136  v.ui = value;
137  int32_t sign = v.si & signC;
138  v.si ^= sign;
139  sign <<= shiftSign;
140  v.si ^= ((v.si + minD) ^ v.si) & -(v.si > subC);
141  v.si ^= ((v.si + maxD) ^ v.si) & -(v.si > maxC);
142  Bits s;
143  s.si = mulC;
144  s.f *= float(v.si);
145  int32_t mask = (norC > v.si) ? -1 : 1;
146  v.si <<= shift;
147  v.si ^= (s.si ^ v.si) & mask;
148  v.si |= sign;
149  return v.f;
150  }
